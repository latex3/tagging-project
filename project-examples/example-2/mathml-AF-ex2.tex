\section{Quadratic Formula}
The solutions to $ax^2+bx+c=0$ are
\[
x=\frac{-b \pm \sqrt{b^2-4ac}}{2a}
\]

\section{Arithmetic}
Absolute values.
\[ \lvert -1\rvert = 1 \]


\section{Matrix Multiplication}
\[
  \begin{pmatrix}
    1&2\\3&4
  \end{pmatrix}
  \begin{pmatrix}
    1&1\\0&1
  \end{pmatrix}
   =
  \begin{pmatrix}
    1&3\\3&7
  \end{pmatrix}
\]

\section{Trigonometric Identities}
\[\sin^2\theta+\cos^2\theta=1\]

\section{Simultaneous Equations}
The equations
\begin{align*}
  2x+y&=3\\
  x-y&=0
\end{align*}
have solution $x=y=1$.
